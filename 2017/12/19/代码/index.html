int main ()
{

    scanf("%lf%lf%lf%lf\n%lf%lf%lf%lf",&x1,&y1,&x2,&y2,&x3,&y3,&x4,&y4);
    printf("%.2f",sqrt((x2-x1)*(x2-x1)+(y2-y1)*(y2-y1)));

字符替换

#include<stdio.h>

int main()
{
    char s1[100], s2[100];
    char ch1, ch2;
    int i, j, n;
    scanf_s("%d", &n);
    for (i = 1;i <= n;i++)
    {
        scanf_s("%s %c %c", s1, &ch1, &ch2);
        for (j = 0;s1[j] != '\0';j++)
        {
            if (s1[j] != ch1)
                s2[j] = s1[j];

}

判断一个数是否为素数

#include<stdio.h>
#include<stdlib.h>

int is_prime(int n)
{
    int k, limit;
    if (2 == n) return 1;
    if (!(n % 2)) return 0;//!（n%2）是当n为偶数是返回值为0，并不是退出程序，而是输出0；
    limit = n / 2;
    for (k = 3;k <= limit;k += 2)
        if (!(n % k)) return 0;//n不能被k整除，当n被k整除输出0；
    return 1;
}
int main()
{
    int a;
    scanf_s("%d", &a);
    printf("%d",is_prime(a));

    getchar();
    system("pause");
    return (0);
}

四个结点的链表

#include<stdio.h>

struct student
{
    char name[20];
    char sex;
    int age;
    struct student *next;
};

int main()
{
    struct student a = { "wang",'M',18 }, b = { "zhang",'F',19 }, c = { "wu",'M',17 }, d = { "cheng",'M',20 };
    struct student *head, *p;
    head = &a;
    a.next = &b;
    b.next = &c;
    c.next = &d;
    d.next = 0;
    p = head;
    printf("&a=%x   &b=%x   &c=%x   &d=%x\n", &a, &b, &c, &d);
    while (p != 0)
    {
        printf("%-12s%3c%3d   ", p->name, p->sex, p->age);
        printf("p=%x  p->next=%x\n", p, p->next);
        p = p->next;
    }

    getchar();
    return 0;
}

螺旋方阵

#include<stdio.h>

int main(void)
{
    int a[21][21];
    int d[2][4] = { 0,1,0,-1,1,0,-1,0 };
    int i, j, k, n, l, dd, ii, jj;
    while (scanf_s("%d", &n) != EOF)
    {
        i = 0;j = -1;k = 1;l = n;dd = 0;
        while (k <= n*n)
        {
            for (ii = 1;ii <= l;ii++)
            {
                i = i + d[0][dd];
                j = j + d[1][dd];
                a[i][j] = k++;
            }
            dd = (dd + 1) % 4;
            if (dd % 2 == 1)
            {
                l--;
            }
        }
        for (i = 0;i < n;i++)
        {
            for (j = 0;j < n;j++)
            {
                printf("%5d", a[i][j]);
            }
            printf("\n");
        }
    }
    getchar();
    system("pause");
    return (0);
}

多项式求和

#include<stdio.h>

int main ()
{
    double a,b,n,n1,n2,m,m1,m2;
    scanf("%lf\n%lf",&a,&b);
    for(n1=1;n1<=a;n1+=2)
        n1+=1/n1;
    for(n2=0;n2<=a;n2+=2)
        n2-=1/n2;
    n=n1+n2;
    for(m1=1;m1<=a;m1+=2)
        m1+=1/m1;
    for(m2=0;m2<=a;m2+=2)
        m2-=1/m2;
    m=m1+m2;
    printf("%.2f\n%.2f",n,m);
    return 0;
}
